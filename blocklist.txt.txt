<script>
let entries = [];
let filter = 'all';

async function load() {
  try {
    const r = await window.storage.get('blocklist-entries');
    if (r && r.value) entries = JSON.parse(r.value);
  } catch(e) { entries = []; }
  render();
}

async function save() {
  try { await window.storage.set('blocklist-entries', JSON.stringify(entries)); } catch(e) {}
}

function validate(val, type) {
  val = val.trim();
  if (!val) return 'Ingresa un valor';
  if (type === 'IP') {
    const ipRx = /^(\d{1,3}\.){3}\d{1,3}$/;
    if (!ipRx.test(val)) return 'IP inválida (ej: 192.168.1.1)';
    const parts = val.split('.');
    if (parts.some(p => parseInt(p) > 255)) return 'IP inválida (valores fuera de rango)';
  }
  if (type === 'CIDR') {
    const cidrRx = /^(\d{1,3}\.){3}\d{1,3}\/\d{1,2}$/;
    if (!cidrRx.test(val)) return 'CIDR inválido (ej: 10.0.0.0/24)';
    const [ip, mask] = val.split('/');
    if (parseInt(mask) > 32) return 'Máscara CIDR inválida (0-32)';
  }
  if (type === 'FQDN') {
    const fqdnRx = /^(\*\.)?([a-zA-Z0-9-]+\.)+[a-zA-Z]{2,}$/;
    if (!fqdnRx.test(val)) return 'FQDN inválido (ej: malware.com o *.bad.com)';
  }
  return null;
}

function showErr(msg) {
  const el = document.getElementById('err-msg');
  el.textContent = msg;
  el.style.display = msg ? 'block' : 'none';
}

function addEntry() {
  const val = document.getElementById('inp-val').value.trim();
  const type = document.getElementById('inp-type').value;
  const label = document.getElementById('inp-label').value.trim();
  const err = validate(val, type);
  if (err) { showErr(err); return; }
  showErr('');
  const dup = entries.find(e => e.val === val);
  if (dup) { showErr('Esta entrada ya existe en la lista'); return; }
  entries.unshift({ id: Date.now(), val, type, label, added: new Date().toISOString().slice(0,10) });
  document.getElementById('inp-val').value = '';
  document.getElementById('inp-label').value = '';
  save();
  render();
}

function deleteEntry(id) {
  entries = entries.filter(e => e.id !== id);
  save();
  render();
}

function setFilter(f, btn) {
  filter = f;
  document.querySelectorAll('.flt').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
  render();
}

function clearAll() {
  if (entries.length === 0) return;
  if (!confirm('¿Eliminar todas las entradas?')) return;
  entries = [];
  save();
  render();
}

function copyOutput() {
  const text = document.getElementById('output-box').textContent;
  navigator.clipboard.writeText(text).then(() => {
    const el = document.getElementById('copied-msg');
    el.style.display = 'inline';
    setTimeout(() => el.style.display = 'none', 2000);
  });
}

function badgeClass(type) {
  if (type === 'IP') return 'badge-ip';
  if (type === 'CIDR') return 'badge-cidr';
  return 'badge-fqdn';
}

function render() {
  const list = document.getElementById('entry-list');
  const outBox = document.getElementById('output-box');
  const visible = filter === 'all' ? entries : entries.filter(e => e.type === filter);
  
  document.getElementById('cnt-total').textContent = entries.length;
  document.getElementById('cnt-ip').textContent = entries.filter(e => e.type === 'IP' || e.type === 'CIDR').length;
  document.getElementById('cnt-fqdn').textContent = entries.filter(e => e.type === 'FQDN').length;

  if (visible.length === 0) {
    list.innerHTML = '<div class="empty"><i class="ti ti-shield-off" style="font-size:24px;display:block;margin-bottom:8px" aria-hidden="true"></i>' + (entries.length === 0 ? 'Sin entradas. Agrega una IP, CIDR o FQDN arriba.' : 'Sin entradas con este filtro.') + '</div>';
  } else {
    list.innerHTML = visible.map(e => `
      <div class="entry">
        <span class="badge ${badgeClass(e.type)}">${e.type}</span>
        <span class="entry-val">${e.val}</span>
        ${e.label ? `<span class="lbl-pill" title="${e.label}">${e.label}</span>` : ''}
        <span style="font-size:11px;color:var(--color-text-secondary);white-space:nowrap">${e.added}</span>
        <button class="btn-del" onclick="deleteEntry(${e.id})" aria-label="Eliminar entrada"><i class="ti ti-x" aria-hidden="true"></i></button>
      </div>
    `).join('');
  }

  if (entries.length === 0) {
    outBox.textContent = '# Sin entradas todavía';
  } else {
    const today = new Date().toISOString().slice(0,10);
    let lines = [`# Blocklist generada: ${today}`, `# Total entradas: ${entries.length}`, ''];
    const ips = entries.filter(e => e.type === 'IP' || e.type === 'CIDR');
    const fqdns = entries.filter(e => e.type === 'FQDN');
    if (ips.length) {
      lines.push('# --- IPs y rangos CIDR ---');
      ips.forEach(e => lines.push(e.label ? `${e.val}  # ${e.label}` : e.val));
      lines.push('');
    }
    if (fqdns.length) {
      lines.push('# --- FQDNs ---');
      fqdns.forEach(e => lines.push(e.label ? `${e.val}  # ${e.label}` : e.val));
    }
    outBox.textContent = lines.join('\n');
  }
}

document.getElementById('inp-val').addEventListener('keydown', e => { if (e.key === 'Enter') addEntry(); });
load();
</script>
